let audioCtx=null, muted=false, bgNode=null, bgGain=null, flipCount=0;


function getCtx(){ if(!audioCtx) audioCtx=new(window.AudioContext||window.webkitAudioContext)(); return audioCtx; }


function toggleMute(){

  muted=!muted;

  document.getElementById('muteBtn').textContent=muted?'🔇':'🔊';

  if(bgGain) bgGain.gain.value=muted?0:0.06;

}


function playSwoosh(fi){

  if(muted) return;

  const ctx=getCtx(), t=ctx.currentTime, bp=200+fi*55;

  const bl=ctx.sampleRate*0.12, buf=ctx.createBuffer(1,bl,ctx.sampleRate), d=buf.getChannelData(0);

  for(let i=0;i<bl;i++) d[i]=(Math.random()*2-1)*(1-i/bl);

  const ns=ctx.createBufferSource(); ns.buffer=buf;

  const nf=ctx.createBiquadFilter(); nf.type='bandpass'; nf.frequency.setValueAtTime(bp*2,t); nf.frequency.exponentialRampToValueAtTime(bp*6,t+0.1); nf.Q.value=1.5;

  const ng=ctx.createGain(); ng.gain.setValueAtTime(0.18,t); ng.gain.exponentialRampToValueAtTime(0.001,t+0.12);

  ns.connect(nf); nf.connect(ng); ng.connect(ctx.destination); ns.start(t); ns.stop(t+0.13);

  const o=ctx.createOscillator(); o.type='sine'; o.frequency.setValueAtTime(bp,t); o.frequency.exponentialRampToValueAtTime(bp*3.5,t+0.08); o.frequency.exponentialRampToValueAtTime(bp*1.2,t+0.18);

  const og=ctx.createGain(); og.gain.setValueAtTime(0.22,t); og.gain.exponentialRampToValueAtTime(0.001,t+0.18);

  o.connect(og); og.connect(ctx.destination); o.start(t); o.stop(t+0.19);

}


function playCashDing(fi){

  if(muted) return;

  const ctx=getCtx(), t=ctx.currentTime;

  const notes=[523,659,784,880,988,1047,1175], freq=notes[Math.min(fi,notes.length-1)];

  [0,0.06,0.12].forEach((delay,i)=>{

    const o=ctx.createOscillator(), g=ctx.createGain();

    o.type=i===2?'triangle':'square'; o.frequency.value=freq*(i===1?1.25:i===2?1.5:1);

    g.gain.setValueAtTime(0,t+delay); g.gain.linearRampToValueAtTime(0.18-i*0.04,t+delay+0.01); g.gain.exponentialRampToValueAtTime(0.001,t+delay+0.22);

    o.connect(g); g.connect(ctx.destination); o.start(t+delay); o.stop(t+delay+0.23);

  });

}


function playBoom(){

  if(muted) return;

  const ctx=getCtx(), t=ctx.currentTime;

  const bl=ctx.sampleRate*0.7, buf=ctx.createBuffer(1,bl,ctx.sampleRate), d=buf.getChannelData(0);

  for(let i=0;i<bl;i++) d[i]=(Math.random()*2-1)*Math.pow(1-i/bl,1.5);

  const src=ctx.createBufferSource(); src.buffer=buf;

  const lpf=ctx.createBiquadFilter(); lpf.type='lowpass'; lpf.frequency.value=180;

  const g=ctx.createGain(); g.gain.setValueAtTime(1.2,t); g.gain.exponentialRampToValueAtTime(0.001,t+0.7);

  src.connect(lpf); lpf.connect(g); g.connect(ctx.destination); src.start(t); src.stop(t+0.71);

  const crack=ctx.createOscillator(); crack.type='sawtooth'; crack.frequency.setValueAtTime(80,t); crack.frequency.exponentialRampToValueAtTime(20,t+0.15);

  const cg=ctx.createGain(); cg.gain.setValueAtTime(0.6,t); cg.gain.exponentialRampToValueAtTime(0.001,t+0.15);

  crack.connect(cg); cg.connect(ctx.destination); crack.start(t); crack.stop(t+0.16);

}


function playFanfare(){

  if(muted) return;

  const ctx=getCtx(), t=ctx.currentTime;

  const melody=[523,659,784,1047,784,1047,1319], dur=[0.1,0.1,0.1,0.15,0.08,0.08,0.35];

  let time=t;

  melody.forEach((freq,i)=>{

    const o=ctx.createOscillator(), g=ctx.createGain();

    o.type='square'; o.frequency.value=freq;

    g.gain.setValueAtTime(0.18,time); g.gain.exponentialRampToValueAtTime(0.001,time+dur[i]);

    o.connect(g); g.connect(ctx.destination); o.start(time); o.stop(time+dur[i]+0.01);

    time+=dur[i]*0.85;

  });

}


function startBgMusic(){

  if(muted) return;

  stopBgMusic();

  const ctx=getCtx();

  bgGain=ctx.createGain(); bgGain.gain.value=0.06; bgGain.connect(ctx.destination);

  const pattern=[130,155,174,196,220,196,174,155]; let step=0, bpm=90;

  function tick(){

    if(!bgNode) return;

    const interval=60/bpm, t=ctx.currentTime;

    const o=ctx.createOscillator(), g=ctx.createGain();

    o.type='square'; o.frequency.value=pattern[step%pattern.length];

    g.gain.setValueAtTime(0.001,t); g.gain.linearRampToValueAtTime(1,t+0.01); g.gain.exponentialRampToValueAtTime(0.001,t+interval*0.7);

    o.connect(g); g.connect(bgGain); o.start(t); o.stop(t+interval*0.75);

    step++; bpm=90+flipCount*8;

    bgNode=setTimeout(tick,interval*900);

  }

  bgNode=setTimeout(tick,100);

}


function stopBgMusic(){ if(bgNode){ clearTimeout(bgNode); bgNode=null; } }


const BASE_VALS=[0.50,0.50,1.00,1.00,1.00,1.50,2.00];

const BOMB_COUNT=2;

let stake=5, board=[], currentWinnings=0, tilesRevealed=0, gameActive=false, bombHitIdx=-1;


function consolation(){ return +(stake*0.20).toFixed(2); }


function fmt(n){

  if(n<1) return `${Math.round(n*100)}¢`;

  if(Math.round(n*100)%100===0) return `$${Math.round(n)}`;

  return `$${n.toFixed(2)}`;

}

function fmtD(n){ return `$${n.toFixed(2)}`; }


function changeStake(){

  stake=parseFloat(document.getElementById('stakeSelect').value);

  document.getElementById('maxWinText').textContent=`$${(stake*1.5).toFixed(2)}`;

  resetGame();

}


function buildBoard(){

  const sc=stake/5;

  let tiles=[...BASE_VALS.map(v=>+(v*sc).toFixed(2)),'bomb','bomb'];

  for(let i=tiles.length-1;i>0;i--){ const j=Math.floor(Math.random()*(i+1)); [tiles[i],tiles[j]]=[tiles[j],tiles[i]]; }

  return tiles;

}


function buildTracker(){

  const sc=stake/5, el=document.getElementById('tracker');

  el.innerHTML='';

  BASE_VALS.map(v=>+(v*sc).toFixed(2)).forEach((v,i)=>{

    const p=document.createElement('div');

    p.dataset.val=v; p.textContent=fmt(v);

    el.appendChild(p);

  });

  for(let i=0;i<BOMB_COUNT;i++){

    const p=document.createElement('div');

    p.className='tracker-pill bomb-pill'; p.id=`pill-b-${i}`;

    p.textContent='💣'; el.appendChild(p);

  }

}

    t.onclick=()=>flipTile(idx);

    t.innerHTML=`<span style="font-size:38px;opacity:0.4;filter:grayscale(1) brightness(2);">👌</span>`;

    g.appendChild(t);

  });

}


function flipTile(idx){

  if(!gameActive) return;

  const t=document.getElementById(`tile-${idx}`);

  if(t.classList.contains('revealed-cash')||t.classList.contains('revealed-bomb-hit')||t.classList.contains('revealed-bomb-other')) return;

  if(flipCount===0) startBgMusic();

  playSwoosh(flipCount);

  t.classList.add('flipping');

  setTimeout(()=>{

    t.classList.remove('flipping');

    const val=board[idx];

    if(val==='bomb'){

      t.innerHTML='<span class="bomb-icon-big">💣</span>';

      t.classList.add('revealed-bomb-hit','shake');

      document.querySelectorAll('.tracker-pill.bomb-pill:not(.bomb-hit)')[0]?.classList.add('bomb-hit');

      bombHitIdx=idx; stopBgMusic();

      setTimeout(()=>playBoom(),80);

      handleBomb();

    } else {

      t.innerHTML=`<span class="tile-val">${fmt(val)}</span>`;

      t.classList.add('revealed-cash');

      currentWinnings=+(currentWinnings+val).toFixed(2);

      tilesRevealed++; flipCount++;

      markPill(val); updateWinnings();

      playCashDing(flipCount-1);

      if(tilesRevealed===BASE_VALS.length) setTimeout(()=>cashOut(),500);

    }

  },160);

}


function markPill(val){

  for(const p of document.querySelectorAll('.tracker-pill:not(.bomb-pill):not(.collected)')){

    if(+p.dataset.val===val){ p.classList.add('collected'); break; }

  }

}


function updateWinnings(){

  const el=document.getElementById('winningsDisplay');

  el.textContent=fmtD(currentWinnings);

  el.classList.add('bump'); setTimeout(()=>el.classList.remove('bump'),220);

  const btn=document.getElementById('cashoutBtn');

  btn.textContent=`Take the ${fmtD(currentWinnings)}`; btn.disabled=currentWinnings===0;

}


function handleBomb(){

  gameActive=false;

  const cons=consolation();

  board.forEach((val,idx)=>{

    const t=document.getElementById(`tile-${idx}`);

    if(idx===bombHitIdx) return;

    if(!t.classList.contains('revealed-cash')){

      if(val==='bomb'){ t.classList.add('revealed-bomb-other'); t.innerHTML='<span class="bomb-icon-sm">💣</span>'; }

      else { t.classList.add('unrevealed-end'); t.innerHTML=`<span class="tile-val dimmed">${fmt(val)}</span>`; }

      t.onclick=null; t.style.cursor='default';

    }

  });

  setTimeout(()=>{

    const sub=document.getElementById('mainSubtitle'), subSm=document.getElementById('mainSubSmall');

    sub.className='subtitle bomb-state'; sub.innerHTML='Oh no, you hit the Bomb!';

    subSm.className='subtitle-small bomb-state'; subSm.innerHTML=`Don't sweat it though, here's ${fmtD(cons)} in Bonus Cash back`;

    const win=document.getElementById('winningsDisplay');

    win.textContent=fmtD(cons); win.classList.add('bomb-won');

    const btn=document.getElementById('cashoutBtn');

    btn.textContent=`Take the ${fmtD(cons)}`; btn.disabled=false; btn.onclick=()=>bombCashOut(cons);

  },500);

}


function bombCashOut(cons){

  launchConfetti(); playFanfare();

  document.getElementById('winAmount').textContent=fmtD(cons);

  document.getElementById('winSubtext').textContent=`You hit a bomb but still walk away with ${fmtD(cons)} in Bonus Cash!`;

  setTimeout(()=>document.getElementById('winOverlay').classList.add('active'),400);

}


function cashOut(){

  if(!gameActive||currentWinnings===0) return;

  gameActive=false; stopBgMusic(); disableGrid();

  document.getElementById('cashoutBtn').disabled=true;

  revealRemaining();

  document.getElementById('winAmount').textContent=fmtD(currentWinnings);

  document.getElementById('winSubtext').textContent=`You avoided the 💣s and left with ${fmtD(currentWinnings)} in Bonus Cash — nice!`;

  launchConfetti(); playFanfare();

  setTimeout(()=>document.getElementById('winOverlay').classList.add('active'),700);

}


function revealRemaining(){

  board.forEach((val,idx)=>{

    const t=document.getElementById(`tile-${idx}`);

    if(!t.classList.contains('revealed-cash')&&!t.classList.contains('revealed-bomb-hit')){

      if(val==='bomb'){ t.classList.add('revealed-bomb-other'); t.innerHTML='<span class="bomb-icon-sm">💣</span>'; }

      else { t.classList.add('unrevealed-end'); t.innerHTML=`<span class="tile-val dimmed">${fmt(val)}</span>`; }

    }

  });

}


function disableGrid(){

  document.querySelectorAll('.tile').forEach(t=>{ t.onclick=null; t.style.cursor='default'; });

}


function launchConfetti(){

  const c=document.getElementById('confettiContainer');

  const cols=['#73FBD3','#FCFF4E','#FF8080','#7742FF','#fff','#2ecc71'];

  for(let i=0;i<90;i++){

    setTimeout(()=>{

      const el=document.createElement('div'); el.className='confetto';

      el.style.cssText=`left:${Math.random()*100}%;top:-20px;background:${cols[~~(Math.random()*cols.length)]};width:${8+Math.random()*8}px;height:${10+Math.random()*10}px;animation-duration:${1.5+Math.random()*2}s`;

      c.appendChild(el); setTimeout(()=>el.remove(),4000);

    },i*25);

  }

}
